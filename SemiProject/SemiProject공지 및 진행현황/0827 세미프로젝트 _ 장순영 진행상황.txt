member_table 생성,

CREATE TABLE coffee.member_table (
  mem_no INT NOT NULL AUTO_INCREMENT,
  mem_id VARCHAR(50) NULL,
  mem_pass VARCHAR(100) NULL,
  mem_name VARCHAR(50) NULL,
  mem_gender VARCHAR(30) NULL,
  mem_birth VARCHAR(50) NULL,
  mem_ph VARCHAR(100) NULL,
  mem_email VARCHAR(100) NULL,
  mem_regdate DATE NULL,
  mem_addr VARCHAR(300) NULL,
  PRIMARY KEY (mem_no));


 movie_table 생성

CREATE TABLE coffee.movie_table (
  mv_no INT NOT NULL AUTO_INCREMENT,
  mv_title VARCHAR(100) NULL,
  mv_poster VARCHAR(100) NULL,
  mv_director VARCHAR(60) NULL,
  mv_st LONGTEXT NULL,
  mv_trailer VARCHAR(100) NULL,
  mv_opendate VARCHAR(30) NULL,
  mv_info VARCHAR(100) NULL,
  mv_regdate DATE NULL,
  mv_genre VARCHAR(60) NULL,
  mv_name VARCHAR(200) NULL,
  mv_adult INT NULL,
  mv_teen INT NULL,
  mv_child INT NULL,
  PRIMARY KEY (mv_no));
